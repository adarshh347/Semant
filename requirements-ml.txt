# Optional ML extras for the Darshan vision pipeline (Track B).
# NOT in core requirements.txt on purpose: these are heavy (~2GB) and the services
# that use them (fashion_clip_service, segmentation_service, fashion_segmentation_service)
# lazy-import and degrade gracefully when absent — so slim API deploys stay slim.
# Install on a box that actually runs the vision pipeline:
#     pip install -r requirements-ml.txt
#
# GPU build (VISION-BUILD-001 B0, 2026-07-19): this box has a CUDA-capable GTX 1650
# (sm_75, driver 595.58.03 / CUDA 13.2). We install the CUDA wheels of torch — the
# linux wheel bundles the CUDA runtime, so NO system CUDA toolkit is required. The
# versions are PINNED and the index is the cu130 channel so a future `pip install -r`
# cannot silently reinstall the CPU wheel (which is what left the box GPU-idle before).
# To roll back to CPU: restore architecture-lab/vision-build-001/B0-pre-cuda-freeze.txt,
# or switch the index below to .../whl/cpu and reinstall. Serverless is still the plan
# for the heavy Phase 2b models (Fashionpedia/SAM2 base+) that a 4GB card can't serve.

--extra-index-url https://download.pytorch.org/whl/cu130
torch==2.13.0
# torchvision must come from the SAME index/build as torch. ultralytics depends on it and
# pip will otherwise pull a mismatched wheel that dies at import with
# "operator torchvision::nms does not exist".
torchvision==0.28.0
transformers
pillow
numpy

# Phase 2a — garment segmentation.
# ultralytics: YOLO11-seg, the non-fashion detector and the graceful fallback.
# opencv: connected components + contour tracing, to turn SegFormer's per-pixel
#         semantic mask into instance polygons.
# The segmenter weights (mattmdjaga/segformer_b2_clothes, ~110MB) download on first use.
ultralytics
opencv-python-headless

# Exact-mask refiner (VISION-BUILD-001 B3): SAM 2.1 Hiera Tiny — benchmarked local-viable
# on the GTX 1650 (peak ~597 MiB VRAM, warm prompts 11-166 ms, clean unload).
# The CHECKPOINT is NOT vendored (147 MB, gitignored under models/). Download once:
#   mkdir -p models && curl -L -o models/sam2.1_hiera_tiny.pt \
#     https://dl.fbaipublicfiles.com/segment_anything_2/092824/sam2.1_hiera_tiny.pt
sam2==1.1.0
